156 /** Internal AES block function.
157   * @param {number array[4]} input array of four 32-bit words to process
158   * @param {boolean} dir false for encryption, true for decryption
159   * @returns {number array[8]} result of the encryption
160   */
161   _aes: function(input, dir)
162   {
163    var key = this.key[(!dir ? 0 : 1) & 1],
164          a = input[0] ^ key[0],
165          b = input[(!dir ? 1 : 3) & 3] ^ key[1],
166          c = input[2] ^ key[2],
167          d = input[(!dir ? 3 : 1) & 3] ^ key[3],
168         a2 = 0, b2 = 0, c2 = 0, i = 0, kIndex = 4,
169        out = [0, 0, 0, 0],
170     rounds = 5+this.keySize,
171      table = this.Stables[(!dir ? 0 : 1 ) & 1],
172         t0 = table[0], t1 = table[1], t2 = table[2],
173         t3 = table[3], sbox = table[4];
174 
175    for(i = 0; i < rounds; i++)
176    {
177     a2 = t0[a>>>24 & 255] ^ t1[b>>16 & 255] ^ t2[c>>8 & 255] ^ t3[d & 255] ^ key[kIndex & 63];
178     b2 = t0[b>>>24 & 255] ^ t1[c>>16 & 255] ^ t2[d>>8 & 255] ^ t3[a & 255] ^ key[(kIndex + 1) & 63];
179     c2 = t0[c>>>24 & 255] ^ t1[d>>16 & 255] ^ t2[a>>8 & 255] ^ t3[b & 255] ^ key[(kIndex + 2) & 63];
180     d  = t0[d>>>24 & 255] ^ t1[a>>16 & 255] ^ t2[b>>8 & 255] ^ t3[c & 255] ^ key[(kIndex + 3) & 63];
181     kIndex += 4; a = a2; b = b2; c = c2;
182    }
183         
184    for(i = 0; i < 4; i++)
185    {
186     out[(!dir ? i : (3&-i)) & 3] =
187     sbox[a>>>24 & 255]<<24 ^ 
188     sbox[b>>16  & 255]<<16 ^
189     sbox[c>>8   & 255]<<8  ^
190     sbox[d      & 255]     ^
191     key[kIndex++ & 63];
192     a2=a; a=b; b=c; c=d; d=a2;
193    }
194 
195    return out;
